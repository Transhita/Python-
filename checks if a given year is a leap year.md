# checks if a given year is a leap year.

## code

   y= int(input("Enter the number: \n"))
                   
   if y% 4 == 0 and (y % 100 == 0 and y % 400) == 0:
                      
             print(f"{y} is a leap year.")
                 
   else:
                    
             print(f"{y} is not a leap year.")
